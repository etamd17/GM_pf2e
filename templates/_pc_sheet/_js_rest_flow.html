// === D15: Themed rest flow === // Replaces native confirm + alert with a "night falls" parchment // modal. Built on the shared .parchment-frame look. The modal is // injected on demand (no hidden DOM baseline cost) and removed on // close. Single CSS transition handles the fade-in; no loops. function _buildRestModal(kind, payload, resolve) { const overlay = document.createElement('div'); overlay.className = 'rest-modal-overlay'; overlay.style.cssText = 'position:fixed;inset:0;background:linear-gradient(180deg, rgba(8,10,20,0.92) 0%, rgba(2,4,10,0.96) 100%);z-index:10000;display:flex;align-items:center;justify-content:center;padding:24px;opacity:0;transition:opacity 0.2s ease;'; const glyph = kind === 'confirm' ? '' : ''; const title = kind === 'confirm' ? 'Night Falls' : 'Dawn Breaks'; const intro = kind === 'confirm' ? 'You make camp and settle in for a long rest. Tending spells, salves, and a warded watch will consume the hours until morning.' : 'The sun crests the ridge. You rise rested — preparations made, spells inscribed, vigor restored.'; const list = kind === 'confirm' ? [ 'HP recovered: Con modifier × level (min 1/level)', 'Focus Points restored', 'Spell slots refreshed (prepared spells cleared)', 'Fatigued, Frightened, Sickened, Stunned cleared', 'Clumsy, Enfeebled, Slowed, Stupefied cleared', 'Wounded removed · Drained reduces by 1', 'Doomed persists · Hero Points unchanged' ] : payload.bullets; overlay.innerHTML = `
${glyph}

${title}

${intro}

    ${list.map(item => `
  • ${item}
    • `).join('')}
${kind === 'confirm' ? '' : ''}
`; document.body.appendChild(overlay); requestAnimationFrame(() => { overlay.style.opacity = '1'; }); function teardown(v) { overlay.style.opacity = '0'; setTimeout(() => overlay.remove(), 180); resolve(v); } const cancel = overlay.querySelector('.rest-cancel'); const confirmBtn = overlay.querySelector('.rest-confirm'); const ack = overlay.querySelector('.rest-ack'); if (cancel) cancel.onclick = () => teardown(false); if (confirmBtn) confirmBtn.onclick = () => teardown(true); if (ack) ack.onclick = () => teardown(true); overlay.addEventListener('click', e => { if (e.target === overlay) teardown(false); }); } function restConfirm() { return new Promise(resolve => _buildRestModal('confirm', null, resolve)); } function restAnnounce(bullets) { return new Promise(resolve => _buildRestModal('announce', { bullets }, resolve)); } async function takeLongRest() { const ok = await restConfirm(); if (!ok) return; try { // Clear all spell state (server handles clearing via long_rest endpoint) window._expendedSlots = {}; window._preparedSpells = {}; window._castPrep = {}; const res = await fetch(`/api/long_rest/${pcNameEncoded}`, { method: 'POST' }); const data = await res.json(); if (data.success) { const r = data.restored; const conSign = r.con_mod >= 0 ? '+' : ''; const hpMsg = r.hp_regained > 0 ? `HP: +${r.hp_regained} recovered (Con ${conSign}${r.con_mod} × level ⇒ ${r.hp_per_level}/lvl) → ${r.hp}/${r.hp_max}` : `HP: already full (${r.hp}/${r.hp_max})`; const bullets = [ hpMsg, `Focus: ${r.focus} restored`, 'Spell slots refreshed' ]; if (r.drained > 0) bullets.push(`Drained: reduced to ${r.drained}`); if (r.doomed > 0) bullets.push(`Doomed: still ${r.doomed} — a night's rest will not lift it`); await restAnnounce(bullets); window.location.reload(); } } catch(e) { console.error('Long rest error:', e); } }
